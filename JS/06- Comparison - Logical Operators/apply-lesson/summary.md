/*
  Comparison Operators
  - == Equal  --> compare Value only 
  - != Not Equal --> compare Value only

  - === Identical --> compare Value && Data type
  - !== Not Identical --> compare Value && Data type

  - > Larger Than
  - >= Larger Than Or Equal

  - < Smaller Than
  - <= Smaller Than Or Equal
*/

/*
  Logical Operators
  - ! Not
  - && And
  - || Or
